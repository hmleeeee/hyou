HTML Structure


//스크립트 적용 전
<div class="ui-scrollbar">
	..........
</div>

//스크립트 적용 후
<div class="ui-scrollbar" scroll-id="uiScrollBar_0">
	<div class="ui-scrollbar-barwrap type-y">
		<button type="button" class="ui-scrollbar-bar" aria-hidden="true" tabindex="-1" data-scrollxy="y" style="...">
			<span class="hide">scroll</span>
		</button>
	</div>
	<div class="ui-scrollbar-barwrap type-x">
		<button type="button" class="ui-scrollbar-bar" aria-hidden="true" tabindex="-1" data-scrollxy="x" style="...">
			<span class="hide">scroll</span>
		</button>
	</div>
	<div class="ui-scrollbar-item ready">
		<div class="ui-scrollbar-wrap">
			..........
		</div>
	</div>
</div>

CSS Style

base style


.ui-scrollbar {position:relative; overflow:auto;}
.ui-scrollbar-item{position:absolute; top:0; left:0; }
.ui-scrollbar.ready {overflow:hidden;}
.ui-scrollbar.ready > .ui-scrollbar-item{width:100%; height:100% !important; overflow:scroll !important;}
.ui-scrollbar.ready .ui-scrollbar-item::-webkit-scrollbar,
.ui-scrollbar.ready .ui-scrollbar::-webkit-scrollbar{width:0; height:0;}
.ie .ui-scrollbar.ready > .ui-scrollbar-item {width:calc(100% + 17px) !important; height:calc(100% + 17px) !important;} 
.ui-scrollbar-barwrap.type-y {position:absolute; top:0; right:0; width:10px; height:100%; background:rgba(0,0,0,0.1); z-index:100;}
.ui-scrollbar-barwrap.type-x {position:absolute; bottom:0; left:0; width:100%; height:10px; background:rgba(0,0,0,0.1); z-index:100;}
.type-y .ui-scrollbar-bar {position:absolute; top:0; right:2px; width:6px; margin:0; height:20px; background:#fff; opacity:1; border-radius:2px;}
.type-x .ui-scrollbar-bar {position:absolute; top:2px; left:0; width:20px; margin:0; height:6px; background:#fff; opacity:1; border-radius:2px;}
.ui-scrollbar:hover .ui-scrollbar-bar {opacity:1; box-shadow:0 0 5px rgba(0,0,0,0.3)}

Javascript

$plugins.uiScrollBar


//전체 적용 .ui-scrollbar
$plugins.uiScrollBar({
	callback: function(){
		console.log('end')
	}
});

//개별 적용
$plugins.uiScrollBar({
	id: 'scroll-id',
	callback: function(){
		console.log('end')
	}
});
옵션 설명
옵션명 유형 기본 설명
id: [string] scroll-id의 값
space: [boolum] true 스크롤의 가로 17px, 세로 17px의 여백
callback: [function] function y축 스크롤이 젤하단에 도착했을 경우 콜백 실행
